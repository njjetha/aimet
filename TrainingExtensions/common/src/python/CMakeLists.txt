        DESTINATION ${CMAKE_INSTALL_LIBDIR}
        PATTERN "__pycache*" EXCLUDE
        PATTERN "CMakeLists.txt" EXCLUDE
        PATTERN "_version.pyi" EXCLUDE
        PATTERN "_gen.py" EXCLUDE
        )
    endif()
